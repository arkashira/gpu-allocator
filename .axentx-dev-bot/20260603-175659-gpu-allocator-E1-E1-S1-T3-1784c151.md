# gpu_allocator.py

import time
from collections import defaultdict

class GPUAllocator:
    def __init__(self):
        self.gpu_inventory = defaultdict(int)
        self.gpu_demand = defaultdict(int)

    def update_inventory(self, gpu_type, count):
        self.gpu_inventory[gpu_type] += count

    def update_demand(self, gpu_type, count):
        self.gpu_demand[gpu_type] += count

    def allocate_gpu(self):
        allocations = {}
        for gpu_type, demand in self.gpu_demand.items():
            available = min(demand, self.gpu_inventory[gpu_type])
            allocations[gpu_type] = available
            self.gpu_inventory[gpu_type] -= available
        return allocations

    def run_allocation_cycle(self):
        while True:
            print("Running allocation cycle...")
            allocations = self.allocate_gpu()
            print(f"Allocations: {allocations}")
            time.sleep(5)  # Simulate real-time updates

# Example usage
if __name__ == "__main__":
    allocator = GPUAllocator()
    allocator.update_inventory('RTX3090', 10)
    allocator.update_demand('RTX3090', 7)
