_handler.setFormatter(logging.Formatter("%(message)s"))
LOGGER.addHandler(_handler)


def _audit(event: str, **payload) -> None:
    """Write a single audit record as a JSON line."""

    LOGGER.info(json.dumps(record))

        self.name = name
        self.description = description

    def __repr__(self) -> str:
        return f"Permission(name={self.name!r})"


class Role:
    """A role aggregates a set of permissions."""
    __slots__ = ("name", "_permissions")

    def __init__(self, name: str, permissions: Optional[List[Permission]] = None) -> None:
        self.name = name
        self._permissions: Dict[str, Permission] = {}
        for p in permissions or []:

    @property

    def has_permission(self, permission_name: str) -> bool:

    def __repr__(self) -> str:
        return f"Role(name={self.name!r}, perms={list(self._permissions)})"
