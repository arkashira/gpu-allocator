        """
        Process an event and detect potential issues.

        :param event: The event to process.
        :return: True if an alert is emitted, False otherwise.
        """

        usage_percent = event['usage_percent']
        location = event['location']

        # Update per-user statistics

        if len(self.user_stats[user_id][1]) > 2:

        if len(distinct_locations) > 2:
            self.alert_service.send_alert(f"Location-hopping detected for user {user_id}")
            return True
