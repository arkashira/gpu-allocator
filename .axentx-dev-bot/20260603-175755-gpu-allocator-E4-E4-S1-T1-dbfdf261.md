package gpuallocator

import (
	"errors"
	"fmt"
	"sync"
)

// GPU represents a physical GPU resource.
type GPU struct {
	ID          string
	TotalMemMB  int
	TotalCores  int
	allocMu     sync.Mutex
	usedMemMB   int
	usedCores   int
}

// GPUAllocator manages a pool of GPUs and allocations.
type GPUAllocator struct {
	gpus map[string]*GPU
	mu   sync.RWMutex
}

// NewGPUAllocator creates an empty allocator.
func NewGPUAllocator() *GPUAllocator {
	return &GPUAllocator{gpus: make(map[string]*GPU)}
}

// RegisterGPU adds a new GPU to the pool.
func (a *GPUAllocator) RegisterGPU(id string, memMB, cores int) error {
	a.mu.Lock()
	defer a.mu.Unlock()
	if _, exists := a.gpus[id]; exists {
		return fmt.Errorf("GPU %s already registered", id)
	}
	a.gpus[id] = &GPU{ID: id, TotalMemMB: memMB, TotalCores: cores}
	return nil
}

// Allocation represents a single allocation request.
type Allocation struct {
	GPUID string // optional – if empty, allocator picks any suitable GPU
	MemMB int
	Cores int
}

// Allocate reserves resources on a GPU that can satisfy the request.
// If req.GPUID is empty, the first GPU that fits is chosen.
// Returns the GPU ID that was allocated or an error if none can satisfy.
func (a *GPUAllocator) Allocate(req Allocation) (string, error) {
	a.mu.RLock()
	defer a.mu.RUnlock()

	// Helper to try a single GPU
	tryGPU := func(g *GPU) bool {
		g.allocMu.Lock()
		defer g.allocMu.Unlock()
		if g.TotalMemMB-g.usedMemMB >= req.MemMB &&
			g.TotalCores-g.usedCores >= req.Cores {
			g.usedMemMB += req.MemMB
			g.usedCores += req.Cores
			return true
		}
		return false
	}

	// If a specific GPU was requested
	if req.GPUID != "" {
		g, ok := a.gpus[req.GPUID]
		if !ok {
			return "", fmt.Errorf("unknown GPU %s", req.GPUID)
		}
		if tryGPU(g) {
			return g.ID, nil
		}
		return "", fmt.Errorf("GPU %s insufficient resources", req.GPUID)
	}

	// Otherwise, scan all GPUs
	for _, g := range a.gpus {
		if tryGPU(g) {
			return g.ID, nil
		}
	}
	return "", errors.New("no GPU has sufficient free resources")
}

// Release frees a previously allocated block.
func (a *GPUAllocator) Release(gpuID string, memMB, cores int) error {
	a.mu.RLock()
	g, ok := a.gpus[gpuID]
	a.mu.RUnlock()
	if !ok {
		return fmt.Errorf("unknown GPU %s", gpuID)
	}
	g.allocMu.Lock()
	defer g.allocMu.Unlock()
	if memMB > g.usedMemMB || cores > g.usedCores {
		return errors.New("release amounts exceed current usage")
	}
	g.usedMemMB -= memMB
	g.usedCores -= cores
	return nil
}

// Status returns a snapshot of all GPUs.
func (a *GPUAllocator) Status() []GPU {
	a.mu.RLock()
	defer a.mu.RUnlock()
	snap := make([]GPU, 0, len(a.gpus))
	for _, g := range a.gpus {
		g.allocMu.Lock()
		snap = append(snap, *g)
		g.allocMu.Unlock()
	}
	return snap
}
