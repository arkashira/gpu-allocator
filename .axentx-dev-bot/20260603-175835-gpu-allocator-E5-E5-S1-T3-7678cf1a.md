"""Data-shaping helpers used by exporters."""

from __future__ import annotations

from datetime import datetime
from typing import Any, Dict, Iterable, List


def flatten_gpu_records(records: Iterable[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Normalise GPU-usage records into a flat, export-friendly list.

    Each input record may contain nested ``gpu`` or ``metrics`` dicts; we
    promote their keys to the top level so CSV/XLSX writers get a clean
    tabular shape.
    """
    flat: List[Dict[str, Any]] = []
    for rec in records:
        row: Dict[str, Any] = {}
        for k, v in rec.items():
            if isinstance(v, dict):
                for sub_k, sub_v in v.items():
                    row[f"{k}.{sub_k}"] = sub_v
            elif isinstance(v, datetime):
                row[k] = v.isoformat()
            else:
                row[k] = v
        flat.append(row)
    return flat


def aggregate_summary(records: Iterable[Dict[str, Any]]) -> Dict[str, Any]:
    """Return a small summary dict suitable for PDF header blocks."""
    records = list(records)
    if not records:
        return {"record_count": 0}

    util_values = [r.get("utilization_pct", r.get("metrics.utilization_pct", 0)) for r in records]
    util_values = [v for v in util_values if isinstance(v, (int, float))]

    return {
        "record_count": len(records),
        "avg_utilization": round(sum(util_values) / len(util_values), 2) if util_values else 0.0,
        "max_utilization": max(util_values) if util_values else 0.0,
        "min_utilization": min(util_values) if util_values else 0.0,
    }
```
