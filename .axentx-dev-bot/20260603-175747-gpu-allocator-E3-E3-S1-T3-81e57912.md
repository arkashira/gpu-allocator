+from collections import defaultdict, deque
+from typing import Dict, List

+class FairnessScheduler:
+    """
+    Implements a weighted round‑robin scheduler for GPU allocation.
+    Each tenant has an associated weight that determines how many
+    allocation cycles it receives relative to others.
+    """
+
+    def __init__(self, weights: Dict[str, int] = None):
+        # Default weight is 1 if not specified
+        self.weights = weights or {}
+        self.queue = deque()
+        self._build_queue()
+
+    def _build_queue(self):
+        """Builds the internal queue based on current weights."""
+        self.queue.clear()
+        for tenant, weight in self.weights.items():
+            for _ in range(weight):
+                self.queue.append(tenant)
+
+    def update_weight(self, tenant: str, weight: int):
+        """Update the weight for a tenant and rebuild the queue."""
+        self.weights[tenant] = weight
+        self._build_queue()
+
+    def next_tenant(self) -> str:
+        """Return the next tenant in round‑robin order."""
+        if not self.queue:
+            raise RuntimeError("FairnessScheduler queue is empty")
+        tenant = self.queue.popleft()
+        self.queue.append(tenant)
+        return tenant
+
+    def get_weights(self) -> Dict[str, int]:
+        return dict(self.weights)
+
+    def set_weights(self, weights: Dict[str, int]):
+        self.weights = weights
+        self._build_queue()
+
+    def get_queue_snapshot(self) -> List[str]:
+        """Return a snapshot of the current queue order."""
+        return list(self.queue)
